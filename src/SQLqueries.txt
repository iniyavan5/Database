CREATE TABLE TRANSFER_TBL (
    Transaction_ID NUMBER(4) PRIMARY KEY,
    Account_Number VARCHAR2(10),
    Beneficiary_Account_Number VARCHAR2(10),
    CONSTRAINT fk_sender_account
        FOREIGN KEY (Account_Number)
        REFERENCES ACCOUNT_TBL (Account_Number),
    CONSTRAINT fk_beneficiary_account
        FOREIGN KEY (Beneficiary_Account_Number)
        REFERENCES ACCOUNT_TBL (Account_Number)
);

alter table TRANSFER_TBL add  Transaction_Date Date;
alter table TRANSFER_TBL add  Transaction_Amount Number(10,2);

CREATE SEQUENCE transactionId
START WITH 1000
